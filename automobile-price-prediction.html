#Funo para normalizar os valores deixando na mesma faixa e podendo identificar com mais facilidade os outliers
def df_norm_valores(df, colunas):
    df_norm = df.copy()

    for coluna in colunas:
        df_norm[coluna] = df[coluna].div(df_norm[coluna].sum(), axis=0)
            
    return df_norm

def identificacao_outlier(df, column_name): 
    #IMPLEMENTAO
    print("Coluna : " + column_name)
    
    Q1 = np.percentile(df[column_name], 25)
    Q3 = np.percentile(df[column_name], 75)

    passo_de_limite = (Q3 - Q1) * 5 
    
    outlier_lower = (Q1 - passo_de_limite)
    outlier_upper = (Q3 + passo_de_limite)

    print("Outliear menor que : {0:0.2f}".format(round((outlier_lower),2)) + " - Quantidade : " + 
              str(df[df[column_name] < outlier_lower][column_name].count()))
    print("Outliear maior que : " + str(outlier_upper) + " - Quantidade : " + 
              str(df[df[column_name] > outlier_upper][column_name].count()))
    print("")

Resposta:

"Modelo de Regresso Linear:" A regresso linear chamada "linear" porque se considera que a relao da resposta s variveis uma funo linear de alguns parmetros. Os modelos de regresso que no so uma funo linear dos parmetros se chamam modelos de regresso no-linear. Sendo uma das primeiras formas de anlise regressiva a ser estudada rigorosamente, e usada extensamente em aplicaes prticas. Isso acontece porque modelos que dependem de forma linear dos seus parmetros desconhecidos, so mais fceis de ajustar que os modelos no-lineares aos seus parmetros, e porque as propriedades estatsticas dos estimadores resultantes so fceis de determinar. Modelos de regresso linear so frequentemente ajustados usando a abordagem dos mnimos quadrados, mas que tambm pode ser montada de outras maneiras, tal como minimizando a "falta de ajuste" em alguma outra norma (com menos desvios absolutos de regresso), ou atravs da minimizao de uma penalizao da verso dos mnimos quadrados. Por outro lado, a abordagem de mnimos quadrados pode ser utilizado para ajustar a modelos que no so modelos lineares. Assim, embora os termos "mnimos quadrados" e "modelo linear" estejam intimamente ligados, eles no so sinnimos.

"Modelo de rvore de Deciso:" O modelo de rvore de Deciso uma representao de uma tabela de deciso sob a forma de uma rvore, porm pode haver outras aplicaes. Tem a mesma utilidade da tabela de deciso. Trata-se de uma maneira alternativa de expressar as mesmas regras que so obtidas quando se constri a tabela. uma abordagem comportamental que usa diagramas para mapear as varias alternativas e resultados de decises de investimento, assim como as probabilidades de ocorrerem. Baseia - se em estimativas e probabilidades associadas aos resultados de cursos de ao que competem entre si. O resultado de cada curso de ao ponderado pela probabilidade associada a ele; o resultado ponderado somado e o valor esperado de cada curso de ao , ento determinado. A alternativa que proporciona o valor esperado mais alto prefervel. Essencialmente, rvores de decises so diagramas que permitem representar e avaliar problemas que envolvem decises sequenciais, colocando em destaque os riscos e os resultados financeiros identificados nos diversos cursos de ao. Para efectuar a representao grfica da rvore de deciso so geralmente usadas linhas para identificar a deciso (por exemplo sim ou no) e ns para identificar a as questes sobre as quais se deve decidir. Cada um dos ramos formado por linhas e ns termina numa espcie de folha que identifica a consequncia mais provvel da sequncia de decises tomadas.

"Modelo de Regresso Ridge:" O Modelo de Regresso Ridge um mtodo de regularizao do modelo que tem como principal objetivo suavizar atributos que sejam relacionados uns aos outros e que aumentam o rudo no modelo (A.K.A multicolinearidade). Com isso com a retirada de determinados atributos do modelo, o mesmo converge para um resultado muito mais estvel em que com a reduo desses atributos, a reduo em termos de acucia do modelo se mantm inalterada. O mecanismo algoritmico que faz isso atravs de um mecanismo de penalizao que coloca um vis e que vai reduzindo os valores os betas at no zero. Com isso os atributos que contribuem menos para o poder preditivo do modelo so levados para a irrelevncia usando esse mecanismo de penalizao do vis.

# funo para reportar as melhores pontuaes no GridSearch
def report_score(results, n_top=3):
    for i in range(1, n_top + 1):
        candidates = np.flatnonzero(results['rank_test_score'] == i)
        for candidate in candidates:
            print("Modelo com classificao: {0}".format(i))
            print("Pontuao de validao mdia: {0:.3f} (std: {1:.3f})".format(
                  results['mean_test_score'][candidate],
                  results['std_test_score'][candidate]))
            print("Parmetros: {0}".format(results['params'][candidate]))
            print("")
